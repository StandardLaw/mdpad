Here's an [example](http://mbostock.github.io/d3/tutorial/bar-1.html) using [D3](http://d3js.org/).

```js

var data = [4, 8, 15, 16, 23, 42];

var x = d3.scale.linear()
    .domain([0, d3.max(data)])
    .range([0, 420]);

var y = d3.scale.ordinal()
    .domain(data)
    .rangeBands([0, 120]);

var chart = d3.select(".jsresult").append("svg")
    .attr("class", "chart")
    .attr("width", 440)
    .attr("height", 200)
    .style("margin-left", "32px") // Tweak alignment¡­
  .append("g")
    .attr("transform", "translate(10,15)");

chart.selectAll("line")
    .data(x.ticks(10))
  .enter().append("line")
    .attr("x1", x)
    .attr("x2", x)
    .attr("y1", 0)
    .attr("y2", 120)
    .style("stroke", "#ccc");

chart.selectAll(".rule")
    .data(x.ticks(10))
  .enter().append("text")
    .attr("class", "rule")
    .attr("x", x)
    .attr("y", 0)
    .attr("dy", -3)
    .attr("text-anchor", "middle")
    .text(String);

chart.selectAll("rect")
    .data(data)
  .enter().append("rect")
    .attr("y", y)
    .attr("width", x)
    .attr("height", y.rangeBand());

chart.selectAll(".bar")
    .data(data)
  .enter().append("text")
    .attr("class", "bar")
    .attr("x", x)
    .attr("y", function(d) { return y(d) + y.rangeBand() / 2; })
    .attr("dx", -3)
    .attr("dy", ".35em")
    .attr("text-anchor", "end")
    .text(String);

chart.append("line")
    .attr("y1", 0)
    .attr("y2", 120)
    .style("stroke", "#000");
```

# NVD3

Here's an [example](http://nvd3.org/ghpages/cumulativeLine.html) using [NVD3](http://nvd3.org/).

```js
var el = $active_element.get(0)
d3.json('data/cumulativeLineData.json', function(data) {
  nv.addGraph(function() {
    var chart = nv.models.cumulativeLineChart()
                  .x(function(d) { return d[0] })
                  .y(function(d) { return d[1]/100 }) //adjusting, 100% is 1.00, not 100 as it is in the data
                  .color(d3.scale.category10().range());

     chart.xAxis
        .tickFormat(function(d) {
            return d3.time.format('%x')(new Date(d))
          });

    chart.yAxis
        .tickFormat(d3.format(',.1%'));

    d3.select(el).append("svg")
        .attr("width", 1000)
        .attr("height", 500)
        .datum(data)
      .transition().duration(500)
        .call(chart);

    nv.utils.windowResize(chart.update);

    return chart;
  });
});
```
